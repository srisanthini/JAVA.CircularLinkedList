import java.util.*;

class Node {
    int data;
    Node next;

    Node(int data) {
        this.data = data;
        this.next = null;
    }
}

class CircularLL {
    Node head = null;

    // Insert at end
    void insert(int data) {
        Node newNode = new Node(data);

        if (head == null) {
            head = newNode;
            head.next = head;
        } else {
            Node temp = head;
            while (temp.next != head) {
                temp = temp.next;
            }
            temp.next = newNode;
            newNode.next = head;
        }
    }

    // Reverse Circular Linked List
    void reverse() {
        if (head == null || head.next == head)
            return;

        Node prev = null;
        Node current = head;
        Node next = null;
        Node last = head;

        // Find last node
        while (last.next != head) {
            last = last.next;
        }

        do {
            next = current.next;
            current.next = prev;
            prev = current;
            current = next;
        } while (current != head);

        head.next = prev;
        head = prev;
    }

    // Split into two halves and print
    void splitAndPrint(int n) {
        int firstHalf;

        if (n % 2 == 0)
            firstHalf = n / 2;
        else
            firstHalf = (n / 2) + 1;

        Node temp = head;

        // First half
        for (int i = 0; i < firstHalf; i++) {
            System.out.print(temp.data + " ");
            temp = temp.next;
        }

        System.out.println();

        // Second half
        for (int i = firstHalf; i < n; i++) {
            System.out.print(temp.data + " ");
            temp = temp.next;
        }
    }

    // Display
    void display() {
        if (head == null)
            return;

        Node temp = head;
        do {
            System.out.print(temp.data + " ");
            temp = temp.next;
        } while (temp != head);
    }
}

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        CircularLL list = new CircularLL();

        int n = sc.nextInt();

        for (int i = 0; i < n; i++) {
            list.insert(sc.nextInt());
        }

        list.reverse();
        list.splitAndPrint(n);
    }
}
